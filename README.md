```python
def parse_lp_text(text):
    # –û—á–∏—Å—Ç–∫–∞ –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö: —É–¥–∞–ª—è–µ–º –ø—É—Å—Ç—ã–µ —Å—Ç—Ä–æ–∫–∏ –∏ –ª–∏—à–Ω–∏–µ –ø—Ä–æ–±–µ–ª—ã
    lines = [l.strip() for l in text.splitlines() if l.strip()]
    
    # –ü–∞—Ä—Å–∏–Ω–≥ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏–∏ (max/min)
    sense = lines[0].lower()
    
    # –ü–∞—Ä—Å–∏–Ω–≥ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–≤ —Ü–µ–ª–µ–≤–æ–π —Ñ—É–Ω–∫—Ü–∏–∏
    c = np.array([float(x) for x in lines[1].split()])
    
    # –ü–∞—Ä—Å–∏–Ω–≥ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–π: –º–∞—Ç—Ä–∏—Ü–∞ A, –∑–Ω–∞–∫–∏ –∏ –ø—Ä–∞–≤—ã–µ —á–∞—Å—Ç–∏
    A, signs, b = [], [], []
    for line in lines[2:]:
        parts = line.split()
        *coeffs, sign, rhs = parts  # –†–∞–∑–¥–µ–ª—è–µ–º –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã, –∑–Ω–∞–∫ –∏ –ø—Ä–∞–≤—É—é —á–∞—Å—Ç—å
        A.append([float(x) for x in coeffs])
        signs.append(sign)
        b.append(float(rhs))
    
    return sense, np.array(c), np.array(A), signs, np.array(b)
```
#### –°–∏–º–ø–ª–µ–∫—Å-–º–µ—Ç–æ–¥
```python
def simplex(T, basis, maximize=True, tol=1e-9, verbose=False, phase_name=""):
    # –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–æ–≤ —Ç–∞–±–ª–∏—Ü—ã (–∏—Å–∫–ª—é—á–∞–µ–º —Å—Ç—Ä–æ–∫—É –æ—Ü–µ–Ω–æ–∫ –∏ —Å—Ç–æ–ª–±–µ—Ü –ø—Ä–∞–≤—ã—Ö —á–∞—Å—Ç–µ–π)
    m = T.shape[0] - 1
    n = T.shape[1] - 1
    it = 0
    
    while True:
        it += 1
        
        # –ê–Ω–∞–ª–∏–∑ –æ—Ü–µ–Ω–æ–∫ –¥–ª—è –≤—ã–±–æ—Ä–∞ –≤–∫–ª—é—á–∞–µ–º–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
        reduced = T[-1, :n]
        if maximize:
            # –î–ª—è –º–∞–∫—Å–∏–º–∏–∑–∞—Ü–∏–∏ –∏—â–µ–º –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ –æ—Ü–µ–Ω–∫–∏
            candidates = np.where(reduced > tol)[0]
            if candidates.size == 0:
                # –í—Å–µ –æ—Ü–µ–Ω–∫–∏ –Ω–µ–ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–µ - –æ–ø—Ç–∏–º—É–º –¥–æ—Å—Ç–∏–≥–Ω—É—Ç
                if verbose:
                    print(f"\n[{phase_name}] –ò—Ç–µ—Ä–∞—Ü–∏—è {it}: –æ–ø—Ç–∏–º—É–º –Ω–∞–π–¥–µ–Ω")
                return "optimal", T, basis
            e = candidates[np.argmax(reduced[candidates])]  # –í—ã–±–æ—Ä –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π —Å max –æ—Ü–µ–Ω–∫–æ–π
        else:
            # –î–ª—è –º–∏–Ω–∏–º–∏–∑–∞—Ü–∏–∏ –∏—â–µ–º –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–µ –æ—Ü–µ–Ω–∫–∏
            candidates = np.where(reduced < -tol)[0]
            if candidates.size == 0:
                if verbose:
                    print(f"\n[{phase_name}] –ò—Ç–µ—Ä–∞—Ü–∏—è {it}: –æ–ø—Ç–∏–º—É–º –Ω–∞–π–¥–µ–Ω")
                return "optimal", T, basis
            e = candidates[np.argmin(reduced[candidates])]  # –í—ã–±–æ—Ä –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π —Å min –æ—Ü–µ–Ω–∫–æ–π

        # –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∏—Å–∫–ª—é—á–∞–µ–º–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π –ø–æ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º—É –æ—Ç–Ω–æ—à–µ–Ω–∏—é
        col = T[:m, e]
        rhs = T[:m, -1]
        with np.errstate(divide="ignore", invalid="ignore"):
            ratio = np.where(col > tol, rhs / col, np.inf)  # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –æ—Ç–Ω–æ—à–µ–Ω–∏–π b_i/a_ie
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –Ω–µ–æ–≥—Ä–∞–Ω–∏—á–µ–Ω–Ω–æ—Å—Ç—å
        if np.all(np.isinf(ratio)):
            print(f"[{phase_name}] –ù–µ–æ–≥—Ä–∞–Ω–∏—á–µ–Ω–Ω–æ—Å—Ç—å –ø–æ –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π {e}")
            return "unbounded", T, basis
        
        l = np.argmin(ratio)  # –í—ã–±–æ—Ä –≤–µ–¥—É—â–µ–π —Å—Ç—Ä–æ–∫–∏

        # –û—Ç–ª–∞–¥–æ—á–Ω—ã–π –≤—ã–≤–æ–¥
        if verbose:
            print(f"\n[{phase_name}] --- –ò—Ç–µ—Ä–∞—Ü–∏—è {it} ---")
            print(f"–í–µ–¥—É—â–∏–π —Å—Ç–æ–ª–±–µ—Ü (entering): {e}")
            print(f"–í–µ–¥—É—â–∞—è —Å—Ç—Ä–æ–∫–∞ (leaving): {l}")
            print("–¢–∞–±–ª–∏—Ü–∞ –¥–æ –ø–∏–≤–æ—Ç–∞:")
            print_tableau(T)
            print("–ë–∞–∑–∏—Å:", basis)

        # –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —à–∞–≥–∞ –∂–æ—Ä–¥–∞–Ω–æ–≤–∞ –∏—Å–∫–ª—é—á–µ–Ω–∏—è
        pivot_on(T, l, e)
        basis[l] = e  # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –±–∞–∑–∏—Å–∞

        if verbose:
            print("–ü–æ—Å–ª–µ –ø–∏–≤–æ—Ç–∞:")
            print_tableau(T)
            print("–ù–æ–≤—ã–π –±–∞–∑–∏—Å:", basis)
```

```python
def two_phase(sense, c, A, signs, b, verbose=False):
    m, n = A.shape

    # ==========================================
    # üîπ –®–∞–≥ 1. –†–∞–∑—Ä–µ—à–∞–µ–º –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ –±—ã—Ç—å –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º–∏
    # ==========================================
    # –ó–∞–º–µ–Ω—è–µ–º –∫–∞–∂–¥—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é x_j –Ω–∞ (x_j‚Å∫ - x_j‚Åª), –≥–¥–µ x_j‚Å∫, x_j‚Åª ‚â• 0
    A_new = np.hstack([A, -A])      # –¥—É–±–ª–∏—Ä—É–µ–º —Å –æ—Ç—Ä–∏—Ü–∞–Ω–∏–µ–º
    c_new = np.hstack([c, -c])      # –∞–Ω–∞–ª–æ–≥–∏—á–Ω–æ –¥–ª—è –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–≤ —Ü–µ–ª–∏
    n = A_new.shape[1]
    A = A_new
    c = c_new

    if verbose:
        print("=== –ü–æ—Å–ª–µ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö (x = x‚Å∫ - x‚Åª) ===")
        print("–ù–æ–≤–∞—è –º–∞—Ç—Ä–∏—Ü–∞ A:")
        print(A)
        print("–ù–æ–≤—ã–π –≤–µ–∫—Ç–æ—Ä c:", c)
        print("==============================================\n")

    # ==========================================
    # üîπ –®–∞–≥ 2. –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è –≤ —Ä–∞–≤–µ–Ω—Å—Ç–≤–∞
    # ==========================================
    A_eq, b_eq = [], []
    for i in range(m):
        if signs[i] == "<=":
            # –¥–æ–±–∞–≤–ª—è–µ–º —Å–ª—ç–∫-–ø–µ—Ä–µ–º–µ–Ω–Ω—É—é
            slack = np.zeros(m)
            slack[i] = 1
            row = np.hstack([A[i], slack])
        elif signs[i] == ">=":
            # –¥–æ–±–∞–≤–ª—è–µ–º "surplus" –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é
            slack = np.zeros(m)
            slack[i] = -1
            row = np.hstack([A[i], slack])
        else:  # "="
            slack = np.zeros(m)
            row = np.hstack([A[i], slack])
        A_eq.append(row)
        b_eq.append(b[i])

    A_eq = np.array(A_eq)
    b_eq = np.array(b_eq)
    n_total = A_eq.shape[1]

    # ==========================================
    # üîπ –®–∞–≥ 3. –§–∞–∑–∞ I (–µ—Å–ª–∏ –Ω—É–∂–Ω—ã –∏—Å–∫—É—Å—Å—Ç–≤–µ–Ω–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ)
    # ==========================================
    need_artificial = any(s in [">=", "="] for s in signs)

    if need_artificial:
        I = np.eye(m)
        A1 = np.hstack([A_eq, I])
        c1 = np.hstack([np.zeros(n_total), np.ones(m)])

        T1 = np.zeros((m + 1, A1.shape[1] + 1))
        T1[:m, :-1] = A1
        T1[:m, -1] = b_eq
        T1[-1, :-1] = c1
        T1[-1, -1] = np.sum(b_eq)
        # –ö–æ—Ä—Ä–µ–∫—Ç–∏—Ä—É–µ–º —Å—Ç—Ä–æ–∫—É —Ü–µ–ª–∏ –¥–ª—è –¥–æ–ø—É—Å—Ç–∏–º–æ–≥–æ –±–∞–∑–∏—Å–∞
        for i in range(m):
            T1[-1, :] -= T1[i, :]

        basis = list(range(n_total, n_total + m))
        status1, T1_final, basis1 = simplex(
            T1, basis, maximize=True, verbose=verbose, phase_name="–§–∞–∑–∞ I"
        )

        if verbose:
            print("\n=== –ö–æ–Ω–µ—Ü –§–∞–∑—ã I ===")
            print("–ë–∞–∑–∏—Å:", basis1)
            print("–¢–∞–±–ª–∏—Ü–∞ –ø–æ—Å–ª–µ –§–∞–∑—ã I:")
            print_tableau(T1_final)
            print("====================\n")

        if abs(T1_final[-1, -1]) > 1e-7:
            return {"status": "no feasible solution"}
    else:
        basis1 = list(range(n, n + m))

    # ==========================================
    # üîπ –®–∞–≥ 4. –§–∞–∑–∞ II ‚Äî –æ—Å–Ω–æ–≤–Ω–∞—è –∑–∞–¥–∞—á–∞
    # ==========================================
    T2 = np.zeros((m + 1, n_total + 1))
    T2[:m, :n_total] = A_eq
    T2[:m, -1] = b_eq
    if sense == "max":
        T2[-1, :len(c)] = c
    else:
        T2[-1, :len(c)] = -c

    status2, T2_final, basis2 = simplex(
        T2, basis1.copy(), maximize=True, verbose=verbose, phase_name="–§–∞–∑–∞ II"
    )

    if verbose:
        print("\n=== –ö–æ–Ω–µ—Ü –§–∞–∑—ã II ===")
        print("–§–∏–Ω–∞–ª—å–Ω–∞—è —Ç–∞–±–ª–∏—Ü–∞:")
        print_tableau(T2_final)
        print("–ë–∞–∑–∏—Å:", basis2)
        print("====================\n")

    if status2 == "unbounded":
        return {"status": "unbounded"}

    # ==========================================
    # üîπ –®–∞–≥ 5. –í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∏—Å—Ö–æ–¥–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ x = x‚Å∫ - x‚Åª
    # ==========================================
    x_full = np.zeros(n_total)
    for i, bi in enumerate(basis2):
        if bi < n_total:
            x_full[bi] = T2_final[i, -1]

    # –ü–µ—Ä–≤–∞—è –ø–æ–ª–æ–≤–∏–Ω–∞ ‚Äî x‚Å∫, –≤—Ç–æ—Ä–∞—è ‚Äî x‚Åª
    half = len(x_full) // 2
    x_orig = x_full[:half] - x_full[half:2 * half]

    z = -T2_final[-1, -1] if sense == "max" else T2_final[-1, -1]

    return {"status": "optimal", "x": x_orig, "z": z}
```
